def get_app_dir():
"""Safely determines the application path for PyInstaller execution."""
if getattr(sys, 'frozen', False):
return os.path.dirname(sys.executable)
return os.path.dirname(os.path.abspath(file))

class CaptureWorker(QThread):
# Emits success (bool) and error message (str)
finished = pyqtSignal(bool, str)

def init(self, devices, paths):
super().init()
self.devices = devices
self.paths = paths

def run(self):
try:
with concurrent.futures.ThreadPoolExecutor(max_workers=2) as executor:
future_to_device = {
executor.submit(capture_and_pull, self.devices[0], self.paths[0]): "Device 1",
executor.submit(capture_and_pull, self.devices[1], self.paths[1]): "Device 2"
}

results = []
for future in concurrent.futures.as_completed(future_to_device):
results.append(future.result())

# If all results are True, emit Success
self.finished.emit(all(results), "")
except Exception as e:
# Catch thread crashes safely
self.finished.emit(False, str(e))

class OcrWorker(QThread):
# Emits text1, text2, and error message
finished = pyqtSignal(str, str, str)

def init(self, path1, path2):
super().init()
self.path1 = path1
self.path2 = path2

def run(self):
try:
with concurrent.futures.ThreadPoolExecutor(max_workers=2) as executor:
future1 = executor.submit(run_ocr_on_image, self.path1)
future2 = executor.submit(run_ocr_on_image, self.path2)

text1 = future1.result()
text2 = future2.result()

self.finished.emit(text1, text2, "")
except Exception as e:
# Tesseract often throws exceptions if not installed properly
self.finished.emit("", "", f"OCR Engine Error: {str(e)}")

def start_capture(self):
try:
devices = get_connected_devices()
if len(devices) < 2:
QMessageBox.warning(self, "Device Error", f"Need exactly 2 devices connected. Found {len(devices)}.")
return

self.lbl_status.setText("Status: Capturing simultaneously...")
self.lbl_status.setStyleSheet("color: blue; font-weight: bold;")

self.worker = CaptureWorker(devices[:2], [self.img1_path, self.img2_path])
self.worker.finished.connect(self.on_capture_finished)
self.worker.start()
except Exception as e:
QMessageBox.critical(self, "ADB Error", f"Failed to initialize ADB capture:\n{e}")

def on_capture_finished(self, success, error_msg):
if success:
self.lbl_status.setText("Status: Capture Complete")
self.lbl_status.setStyleSheet("color: green; font-weight: bold;")
self.update_preview(self.img1_path, self.lbl_img1)
self.update_preview(self.img2_path, self.lbl_img2)
self.update_combined_preview()
else:
self.lbl_status.setText("Status: Capture Failed")
self.lbl_status.setStyleSheet("color: red; font-weight: bold;")
if error_msg:
QMessageBox.critical(self, "Capture Thread Crash", error_msg)

def update_preview(self, path, label_widget):
try:
if not os.path.exists(path):
return

# Ensure label size is valid before scaling (prevents divide by zero UI crashes)
lbl_w = max(1, label_widget.width())
lbl_h = max(1, label_widget.height())

pixmap = QPixmap(path).scaled(lbl_w, lbl_h, Qt.KeepAspectRatio, Qt.SmoothTransformation)
if not pixmap.isNull():
label_widget.setPixmap(pixmap)
else:
label_widget.setText("Image corrupted or unreadable.")
except Exception as e:
print(f"Error loading preview: {e}")

def trigger_crop(self, path, label_widget):
if not os.path.exists(path):
QMessageBox.warning(self, "Error", "No image found. Capture first.")
return

self.lbl_status.setText("Status: Draw box in OpenCV window and press ENTER (or C to cancel)")
try:
success = open_opencv_cropper(path)
if success:
self.update_preview(path, label_widget)
self.update_combined_preview()
self.lbl_status.setText("Status: Cropped successfully")
else:
self.lbl_status.setText("Status: Crop cancelled")
except Exception as e:
self.lbl_status.setText("Status: OpenCV Error")
QMessageBox.critical(self, "Cropping Error", f"Failed to open cropper:\n{e}")

def update_combined_preview(self):
try:
self.combined_cv2_img = get_combined_image_cv2(self.img1_path, self.img2_path)

if self.combined_cv2_img is not None:
h, w, ch = self.combined_cv2_img.shape
bytes_per_line = ch * w
rgb_image = cv2.cvtColor(self.combined_cv2_img, cv2.COLOR_BGR2RGB)

q_img = QImage(rgb_image.data, w, h, bytes_per_line, QImage.Format_RGB888)

# Prevent scaling crash if widget isn't fully drawn yet
lbl_w = max(1, self.lbl_combined.width())
lbl_h = max(1, self.lbl_combined.height())

pixmap = QPixmap.fromImage(q_img).scaled(lbl_w, lbl_h, Qt.KeepAspectRatio, Qt.SmoothTransformation)
self.lbl_combined.setPixmap(pixmap)
except Exception as e:
QMessageBox.warning(self, "Preview Error", f"Failed to combine images:\n{e}")

def prompt_save(self):
if self.combined_cv2_img is None:
QMessageBox.warning(self, "Error", "No combined image to save. Capture and crop first.")
return

dialog = QDialog(self)
dialog.setWindowTitle("Save Configuration")
layout = QFormLayout(dialog)

name_input = QLineEdit()
id_input = QLineEdit()

layout.addRow("Name (Mandatory):", name_input)
layout.addRow("ID (Mandatory):", id_input)

btn_confirm = QPushButton("Confirm & Save")
layout.addRow(btn_confirm)

def on_confirm():
name = name_input.text().strip()
user_id = id_input.text().strip()

if not name or not user_id:
QMessageBox.warning(dialog, "Validation Error", "BOTH Name and ID are mandatory.")
return

dialog.accept()
self.execute_save_and_ocr(name, user_id)

btn_confirm.clicked.connect(on_confirm)
dialog.exec_()

def execute_save_and_ocr(self, name, user_id):
self.lbl_status.setText("Status: Saving image and processing OCR...")
self.lbl_status.setStyleSheet("color: blue; font-weight: bold;")

try:
# Save Image to disk and get base filename
saved_path, base_filename = add_watermark_and_save(
self.combined_cv2_img, name, user_id, self.out_dir
)

# Start OCR asynchronously
self.ocr_worker = OcrWorker(self.img1_path, self.img2_path)
# Pass saved_path and base_filename using lambda
self.ocr_worker.finished.connect(
lambda t1, t2, err: self.on_ocr_finished(t1, t2, err, saved_path, base_filename)
)
self.ocr_worker.start()

except Exception as e:
self.lbl_status.setText("Status: Save Failed")
self.lbl_status.setStyleSheet("color: red; font-weight: bold;")
QMessageBox.critical(self, "Save Error", f"Failed to save image:\n{traceback.format_exc()}")

def on_ocr_finished(self, text1, text2, error_msg, saved_path, base_filename):
if error_msg:
self.lbl_status.setText("Status: Image Saved, but OCR failed.")
QMessageBox.warning(self, "OCR Warning", error_msg)
return

try:
# Write OCR results to text file
txt_path = os.path.join(self.out_dir, f"{base_filename}_OCR.txt")
with open(txt_path, "w", encoding="utf-8") as f:
f.write("--- OCR Image 1 ---\n")
f.write(text1 + "\n\n")
f.write("--- OCR Image 2 ---\n")
f.write(text2 + "\n")

self.lbl_status.setText(f"Status: Saved Image and OCR Text to {self.out_dir}")
self.lbl_status.setStyleSheet("color: green; font-weight: bold;")

# Show Success Dialog
msg = QMessageBox(self)
msg.setWindowTitle("Process Complete")
msg.setText("Image and OCR Text Saved Successfully!")
msg.setDetailedText(f"Saved to: {self.out_dir}\n\n--- OCR Image 1 ---\n{text1}\n\n--- OCR Image 2 ---\n{text2}")
msg.exec_()

except Exception as e:
QMessageBox.critical(self, "File Error", f"Failed to write text file:\n{e}")
