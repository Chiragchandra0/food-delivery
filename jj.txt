def resource_path(relative_path):
    try:
        if getattr(sys, 'frozen', False):
            base_path = sys._MEIPASS
        else:
            base_path = os.path.abspath(".")
        return os.path.join(base_path, relative_path)
    except Exception:
        return relative_path

def run_adb(*args):
    try:
        startupinfo = subprocess.STARTUPINFO()
        startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW
        startupinfo.wShowWindow = subprocess.SW_HIDE

        result = subprocess.check_output(
            [ADB_PATH, *args],
            startupinfo=startupinfo,
            stderr=subprocess.DEVNULL,
            timeout=10
        ).decode().strip()
        return result
    except Exception:
        return "Not Found"

def get_eid() -> str:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        result = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#06#'; "
            "uiautomator dump --compressed /dev/stdout | grep -oE '[0-9]{32}'")
        eid = result.strip()
        return eid if eid else "Not Found"
    except Exception:
        return "Not Found"

def get_serial() -> str:
    try:
        return run_adb("shell", "getprop", "ro.serialno") or " "
    except Exception:
        return "Not Found"

arr = [] ### array which contain the values for version, rf cal etc.
def get_version() -> str:
    global arr
    arr.clear()
    try:
        text = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#12580*369#';",
            " uiautomator dump --compressed /dev/stdout | grep string")
        version = text

        if not version or version == "Not Found":
            arr.extend(["N/A", "N/A", "N/A", "N/A", "N/A"])
            return "Not Found"

        content = version.split()
        a = content[50].strip('"') if len(content) > 50 else "N/A"
        b = content[72].strip('"') if len(content) > 72 else "N/A"
        c = content[94].strip('"') if len(content) > 94 else "N/A"
        d = content[117].strip('"') if len(content) > 117 else "N/A"
        e = content[140].strip('"') if len(content) > 140 else "N/A"
        arr.extend([a, b, c, d, e])
        return "Success"
    except Exception:
        arr.extend(["Error", "Error", "Error", "Error", "Error"])
        return "Error"

def get_imei() -> str:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        result = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#06#'; "
            "uiautomator dump --compressed /dev/stdout | grep -oE '[0-9]{15}'")
        imei = result.strip()
        return imei[:15] if imei else "Not Found"
    except Exception:
        return "Not Found"

def fetch_device_info() -> dict:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        get_version()
        serial       = run_adb("get-serialno")          or "N/A"
        brand        = run_adb("shell", "getprop", "ro.product.brand")          or "N/A"
        model        = run_adb("shell", "getprop", "ro.product.model")          or "N/A"
        android_ver  = run_adb("shell", "getprop", "ro.build.version.release")  or "N/A"
        arr.append(model) #5th index
        
        battery = "N/A"
        battery_dump = run_adb("shell", "dumpsys", "battery")
        if battery_dump != "Not Found":
            for ln in battery_dump.splitlines():
                if "level:" in ln:
                    battery = ln.split(":")[-1].strip() + " %"
                    break

        wifi_mac = "N/A"
        ip_dump = run_adb("shell", "ip", "addr", "show", "wlan0")
        if ip_dump != "Not Found":
            for ln in ip_dump.splitlines():
                if "link/ether" in ln:
                    wifi_mac = ln.split()[1]
                    break

        bt_mac = (
            run_adb("shell", "settings", "get", "secure", "bluetooth_address") or "N/A"
        )

        return {
            "Serial No": serial,
            "Brand": brand.upper() if brand != "N/A" else "N/A",
            "Model": model,
            "Android Version": android_ver,
            "Battery": battery,
            "WiFi MAC": wifi_mac.upper() if wifi_mac != "N/A" else "N/A",
            "Bluetooth MAC": bt_mac.upper() if bt_mac != "N/A" else "N/A",
            "Base Version": arr[0] if len(arr) > 0 else "N/A",
            "RF Cal": arr[3] if len(arr) > 3 else "N/A",
            "HW Rev": arr[4] if len(arr) > 4 else "N/A"
        }
    except Exception as e:
        return {"Error": f"Failed fetching info: {str(e)}"}

    def update_final_result(self):
        statuses = {
            "Binary": self.binary_status,
            "UI": self.ui_status,
            "EID": self.eid_status
        }
        total_tests = len(statuses)
        completed_tests = sum(1 for status in statuses.values() if status is not None)
        passed_tests = sum(1 for status in statuses.values() if status is True)

        if completed_tests == total_tests:
            if passed_tests == total_tests:
                self.final_result_label.configure(text="PASS", text_color="#1A7A1A")
            else:
                self.final_result_label.configure(text="FAIL", text_color="#BB0000")
        elif completed_tests > 0:
            self.final_result_label.configure(
                text=f"Pending... ({completed_tests}/{total_tests} tests complete)",
                text_color="#333333"
            )
        else:
            self.final_result_label.configure(text="Pending...", text_color="#333333")

    def extract_ui_images_from_excel(self, excel_file):
        try:
            from openpyxl import load_workbook
            os.makedirs("ref_img", exist_ok=True)
            for f in glob("ref_img/*.png"):
                try: os.remove(f)
                except Exception: pass

            workbook = load_workbook(excel_file)
            sheet = workbook.active
            images_info = []

            for img in sheet._images:
                if hasattr(img, 'anchor') and hasattr(img.anchor, '_from'):
                    cell = img.anchor._from
                    images_info.append((cell.row, cell.col, img))

            images_info.sort(key=lambda x: (x[0], x[1]))
            extracted_images = []
            
            for idx, (row, col, img) in enumerate(images_info):
                image_path = os.path.join("ref_img", f"extracted_image_{idx + 1}.png")
                with open(image_path, "wb") as img_file:
                    img_file.write(img._data())
                extracted_images.append(image_path)
            return extracted_images

        except zipfile.BadZipFile:
            messagebox.showerror("File Error", "File is not a valid zip archive (not a true .xlsx).")
            return []
        except Exception as e:
            messagebox.showerror("Error", f"Failed to extract images: {str(e)}")
            return []

    def compare_ui_images(self):
        excel_file = self.ui_file_path.get()
        if not excel_file:
            self.after(0, lambda: messagebox.showerror("Error", "Please select an Excel file for UI verification"))
            self.after(0, lambda: self._set_ui_busy(False))
            return

        try:
            import cv2
            ref_images = self.extract_ui_images_from_excel(excel_file)
            if len(ref_images) < 4:
                self.after(0, lambda: messagebox.showerror("Error", f"Extracted only {len(ref_images)} images, need 4"))
                self.after(0, lambda: self._set_ui_busy(False))
                return

            if not self.capture_device_images():
                self.after(0, lambda: self._set_ui_busy(False))
                return

            target_images = sorted(glob(os.path.join(os.path.dirname(os.path.abspath(__file__)),'target_img/*.*')))
            if len(target_images) < 4:

    def compare_launcher_ui(self, ref, target, similarity_thresh=0.70, position_tol=80):
        try:
            import cv2
            if ref is None or target is None:
                raise ValueError("Image not loaded")

            ref_h, ref_w = ref.shape[:2]
            tgt_h, tgt_w = target.shape[:2]
            scale_x = tgt_w / ref_w
            scale_y = tgt_h / ref_h

            result_img = target.copy()
            gray_ref = cv2.cvtColor(ref, cv2.COLOR_BGR2GRAY)
            thresh = cv2.adaptiveThreshold(gray_ref, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 11, 2)
            contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

            icon_boxes = []
            for cnt in contours:
                x, y, w, h = cv2.boundingRect(cnt)
                if h == 0: continue
                ratio = w / float(h)
                if (0.7 < ratio < 1.3 and 20 < w < 120 and 20 < h < 120):
                    icon_boxes.append((x, y, w, h))

            icon_boxes = sorted(icon_boxes, key=lambda b: (b[1], b[0]))
            orb = cv2.SIFT_create(nfeatures=3000)

            missing_apps = []
            misplaced_apps = []

            for idx, (x, y, w, h) in enumerate(icon_boxes):
                ref_icon = ref[y:y + h, x:x + w]
                kp1, des1 = orb.detectAndCompute(ref_icon, None)
                if des1 is None: continue

                expected_x = int(x * scale_x)
                expected_y = int(y * scale_y)
                search_margin = 150
                sx1 = max(0, expected_x - search_margin)
                sy1 = max(0, expected_y - search_margin)
                sx2 = min(tgt_w, expected_x + int(w * scale_x) + search_margin)
                sy2 = min(tgt_h, expected_y + int(h * scale_y) + search_margin)

                search_roi = target[sy1:sy2, sx1:sx2]
                if search_roi.size == 0: continue

                ref_resized = cv2.resize(ref_icon, (max(10, int(w * scale_x)), max(10, int(h * scale_y))))
                match = cv2.matchTemplate(search_roi, ref_resized, cv2.TM_CCOEFF_NORMED)
                _, max_val, _, max_loc = cv2.minMaxLoc(match)

                if max_val < similarity_thresh:
                    missing_apps.append({"icon_id": idx, "expected": (expected_x, expected_y)})
                    cv2.rectangle(result_img, (expected_x, expected_y), (expected_x + int(w * scale_x), expected_y + int(h * scale_y)), (0, 0, 255), 3)
                    cv2.putText(result_img, "MISSING", (expected_x, expected_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
                    continue

                found_x = sx1 + max_loc[0]
                found_y = sy1 + max_loc[1]
                if abs(found_x - expected_x) > position_tol or abs(found_y - expected_y) > position_tol:
                    misplaced_apps.append({"icon_id": idx, "expected": (expected_x, expected_y), "found": (found_x, found_y)})
                    cv2.rectangle(result_img, (found_x, found_y), (found_x + ref_resized.shape[1], found_y + ref_resized.shape[0]), (0, 165, 255), 3)
                    cv2.line(result_img, (expected_x, expected_y), (found_x, found_y), (0, 165, 255), 2)
                    cv2.putText(result_img, "MISPLACED", (found_x, found_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 165, 255), 2)

            return result_img, missing_apps, misplaced_apps

        except Exception as e:
            print(f"Error in compare_launcher_ui: {str(e)}")
            return target.copy() if target is not None else None, [], []

    def _on_fetch(self):
        inputs = {key: self._entries[key].get().strip() for key, *_ in [(k,) for _, k, _ in self.FIELDS]}

        if inputs['eid'] and len(inputs['eid']) != 32:
            messagebox.showerror("Invalid EID", "EID must be 32 characters long")
            return
        if inputs['sn'] and len(inputs['sn']) != 11:
            messagebox.showerror("Invalid S/N", "S/N must be 11 characters long")
            return
        if inputs['imei'] and len(inputs['imei']) != 15:
            messagebox.showerror("Invalid IMEI", "IMEI must be 15 characters long")
            return
        if not any(inputs.values()):
            messagebox.showwarning("Input Required", "Enter at least one value to compare.")
            return

        self._result_lbl.configure(text="Fetching…", text_color="#333333")
        self._set_ui_busy(True)

        def worker():
            verdicts: list[bool] = []
            for _, key, fn in self.FIELDS:
                scanned = self._entries[key].get().strip()
                if not scanned:
                    self.after(0, lambda k=key: (
                        self._fetched_lbls[k].configure(text="— — —"),
                        self._set_status(k, "idle")
                    ))
                    continue

                fetched = fn()
                match = (scanned == fetched) and fetched != "Not Found"
                verdicts.append(match)

                status = "pass" if match else "fail"
                self.after(0, lambda k=key, v=fetched, s=status: (
                    self._fetched_lbls[k].configure(text=v),
                    self._set_status(k, s),
                ))

            self.after(0, lambda: self._finish_compare(verdicts))

        threading.Thread(target=worker, daemon=True).start()

    def _finish_compare(self, verdicts: list[bool]):
        if not verdicts:
            self._result_lbl.configure(text="PASS/FAIL", text_color="black")
        elif all(verdicts):
            self.eid_status=True
            self.update_final_result()
            self._result_lbl.configure(text="✅  PASS", text_color="#1A7A1A")
        else:
            self.eid_status=False
            self.update_final_result()
            self._result_lbl.configure(text="❌  FAIL", text_color="#BB0000")
        self._set_ui_busy(False)

def run_adb(*args):
    try:
        startupinfo = subprocess.STARTUPINFO()
        startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW
        startupinfo.wShowWindow = subprocess.SW_HIDE

        result = subprocess.check_output(
            [ADB_PATH, *args],
            startupinfo=startupinfo,
            stderr=subprocess.DEVNULL,
            timeout=10
        ).decode().strip()
        return result
    except Exception:
        return "Not Found"

def get_eid() -> str:

        result = run_adb("shell",

arr = [] ### array which contain the values for version, rf cal etc.
def get_version() -> str:
    global arr
    arr.clear()
    try:
        text = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#12580*369#';",
            " uiautomator dump --compressed /dev/stdout | grep string")
        version = text

        if not version or version == "Not Found":
            arr.extend(["N/A", "N/A", "N/A", "N/A", "N/A"])
            return "Not Found"

        content = version.split()
        a = content[50].strip('"') if len(content) > 50 else "N/A"
        b = content[72].strip('"') if len(content) > 72 else "N/A"
        c = content[94].strip('"') if len(content) > 94 else "N/A"
        d = content[117].strip('"') if len(content) > 117 else "N/A"
        e = content[140].strip('"') if len(content) > 140 else "N/A"
        arr.extend([a, b, c, d, e])
        return "Success"
    except Exception:
        arr.extend(["Error", "Error", "Error", "Error", "Error"])
        return "Error"

def get_imei() -> str:
    try:

        result = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#06#'; "
            "uiautomator dump --compressed /dev/stdout | grep -oE '[0-9]{15}'")
        imei = result.strip()
        return imei[:15] if imei else "Not Found"
    except Exception:
        return "Not Found"

def fetch_device_info() -> dict:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        get_version()
        serial       = run_adb("get-serialno")          or "N/A"
        brand        = run_adb("shell", "getprop", "ro.product.brand")          or "N/A"
        model        = run_adb("shell", "getprop", "ro.product.model")          or "N/A"
        android_ver  = run_adb("shell", "getprop", "ro.build.version.release")  or "N/A"
        arr.append(model) #5th index
        
        battery = "N/A"
        battery_dump = run_adb("shell", "dumpsys", "battery")
        if battery_dump != "Not Found":
            for ln in battery_dump.splitlines():
                if "level:" in ln:
                    battery = ln.split(":")[-1].strip() + " %"
                    break

        wifi_mac = "N/A"
        ip_dump = run_adb("shell", "ip", "addr", "show", "wlan0")
        if ip_dump != "Not Found":
            for ln in ip_dump.splitlines():
                if "link/ether" in ln:
                    wifi_mac = ln.split()[1]
                    break

        bt_mac = (
            run_adb("shell", "settings", "get", "secure", "bluetooth_address") or "N/A"
        )

        return {
            "Serial No": serial,
            "Brand": brand.upper() if brand != "N/A" else "N/A",
            "Model": model,
            "Android Version": android_ver,
            "Battery": battery,
            "WiFi MAC": wifi_mac.upper() if wifi_mac != "N/A" else "N/A",
            "Bluetooth MAC": bt_mac.upper() if bt_mac != "N/A" else "N/A",
            "Base Version": arr[0] if len(arr) > 0 else "N/A",
            "RF Cal": arr[3] if len(arr) > 3 else "N/A",
            "HW Rev": arr[4] if len(arr) > 4 else "N/A"
        }
    except Exception as e:
        return {"Error": f"Failed fetching info: {str(e)}"}

    def extract_ui_images_from_excel(self, excel_file):
        try:
            from openpyxl import load_workbook
            os.makedirs("ref_img", exist_ok=True)
            for f in glob("ref_img/*.png"):
                try: os.remove(f)
                except Exception: pass

            workbook = load_workbook(excel_file)
            sheet = workbook.active
            images_info = []

            for img in sheet._images:
                if hasattr(img, 'anchor') and hasattr(img.anchor, '_from'):
                    cell = img.anchor._from
                    images_info.append((cell.row, cell.col, img))

            images_info.sort(key=lambda x: (x[0], x[1]))
            extracted_images = []
            
            for idx, (row, col, img) in enumerate(images_info):
                image_path = os.path.join("ref_img", f"extracted_image_{idx + 1}.png")
                with open(image_path, "wb") as img_file:
                    img_file.write(img._data())
                extracted_images.append(image_path)
            return extracted_images

        except zipfile.BadZipFile:
            messagebox.showerror("File Error", "File is not a valid zip archive (not a true .xlsx).")
            return []
        except Exception as e:
            messagebox.showerror("Error", f"Failed to extract images: {str(e)}")
            return []

    def compare_ui_images(self):
        excel_file = self.ui_file_path.get()
        if not excel_file:
            self.after(0, lambda: messagebox.showerror("Error", "Please select an Excel file for UI verification"))
            self.after(0, lambda: self._set_ui_busy(False))
            return

        try:
            import cv2
            ref_images = self.extract_ui_images_from_excel(excel_file)
            if len(ref_images) < 4:
                self.after(0, lambda: messagebox.showerror("Error", f"Extracted only {len(ref_images)} images, need 4"))
                self.after(0, lambda: self._set_ui_busy(False))
                return

            if not self.capture_device_images():
                self.after(0, lambda: self._set_ui_busy(False))
                return

            target_images = sorted(glob(os.path.join(os.path.dirname(os.path.abspath(__file__)),'target_img/*.*')))
            if len(target_images) < 4:
                self.after(0, lambda: messagebox.showerror("Error", f"Captured only {len(target_images)} screenshots"))
                self.after(0, lambda: self._set_ui_busy(False))
                return

            device_model = run_adb("shell", "getprop", "ro.product.model")
            if device_model == "Not Found": device_model = "UnknownDevice"
            os.makedirs(device_model, exist_ok=True)

            self.comparison_images = []
            results = []
            all_pass = True

            for i in range(min(4, len(ref_images), len(target_images))):
                try:
                    ref_img = cv2.imread(ref_images[i])
                    target_img = cv2.imread(target_images[i])
                    if ref_img is None or target_img is None:
                        raise ValueError("Failed to load image via cv2")

                    result_img, missing_apps, misplaced_apps = self.compare_launcher_ui(ref_img, target_img)
                    output_path = os.path.join(device_model, f"comparison_result_{i + 1}.png")
                    cv2.imwrite(output_path, result_img)
                    self.comparison_images.append(output_path) 

                    if missing_apps or misplaced_apps:
                        results.append(f"Image {i + 1}: FAIL - Missing: {len(missing_apps)}, Misplaced: {len(misplaced_apps)}")
                        all_pass = False
                    else:
                        results.append(f"Image {i + 1}: PASS")
                except Exception as e:
                    results.append(f"Image {i + 1}: Error - {str(e)}")
                    all_pass = False

    def compare_launcher_ui(self, ref, target, similarity_thresh=0.70, position_tol=80):
        try:
            import cv2
            if ref is None or target is None:
                raise ValueError("Image not loaded")

            ref_h, ref_w = ref.shape[:2]
            tgt_h, tgt_w = target.shape[:2]
            scale_x = tgt_w / ref_w
            scale_y = tgt_h / ref_h

            result_img = target.copy()
            gray_ref = cv2.cvtColor(ref, cv2.COLOR_BGR2GRAY)
            thresh = cv2.adaptiveThreshold(gray_ref, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 11, 2)
            contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

            icon_boxes = []
            for cnt in contours:
                x, y, w, h = cv2.boundingRect(cnt)
                if h == 0: continue
                ratio = w / float(h)
                if (0.7 < ratio < 1.3 and 20 < w < 120 and 20 < h < 120):
                    icon_boxes.append((x, y, w, h))

            icon_boxes = sorted(icon_boxes, key=lambda b: (b[1], b[0]))
            orb = cv2.SIFT_create(nfeatures=3000)

            missing_apps = []
            misplaced_apps = []

            for idx, (x, y, w, h) in enumerate(icon_boxes):
                ref_icon = ref[y:y + h, x:x + w]
                kp1, des1 = orb.detectAndCompute(ref_icon, None)
                if des1 is None: continue

                expected_x = int(x * scale_x)
                expected_y = int(y * scale_y)
                search_margin = 150
                sx1 = max(0, expected_x - search_margin)
                sy1 = max(0, expected_y - search_margin)
                sx2 = min(tgt_w, expected_x + int(w * scale_x) + search_margin)
                sy2 = min(tgt_h, expected_y + int(h * scale_y) + search_margin)

                search_roi = target[sy1:sy2, sx1:sx2]
                if search_roi.size == 0: continue

                ref_resized = cv2.resize(ref_icon, (max(10, int(w * scale_x)), max(10, int(h * scale_y))))
                match = cv2.matchTemplate(search_roi, ref_resized, cv2.TM_CCOEFF_NORMED)
                _, max_val, _, max_loc = cv2.minMaxLoc(match)

                if max_val < similarity_thresh:
                    missing_apps.append({"icon_id": idx, "expected": (expected_x, expected_y)})
                    cv2.rectangle(result_img, (expected_x, expected_y), (expected_x + int(w * scale_x), expected_y + int(h * scale_y)), (0, 0, 255), 3)
                    cv2.putText(result_img, "MISSING", (expected_x, expected_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
                    continue

                found_x = sx1 + max_loc[0]
                found_y = sy1 + max_loc[1]
                if abs(found_x - expected_x) > position_tol or abs(found_y - expected_y) > position_tol:
                    misplaced_apps.append({"icon_id": idx, "expected": (expected_x, expected_y), "found": (found_x, found_y)})
                    cv2.rectangle(result_img, (found_x, found_y), (found_x + ref_resized.shape[1], found_y + ref_resized.shape[0]), (0, 165, 255), 3)
                    cv2.line(result_img, (expected_x, expected_y), (found_x, found_y), (0, 165, 255), 2)
                    cv2.putText(result_img, "MISPLACED", (found_x, found_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 165, 255), 2)

            return result_img, missing_apps, misplaced_apps

        except Exception as e:
            print(f"Error in compare_launcher_ui: {str(e)}")
            return target.copy() if target is not None else None, [], []

    def _on_fetch(self):
        inputs = {key: self._entries[key].get().strip() for key, *_ in [(k,) for _, k, _ in self.FIELDS]}

        if inputs['eid'] and len(inputs['eid']) != 32:
            messagebox.showerror("Invalid EID", "EID must be 32 characters long")
            return
        if inputs['sn'] and len(inputs['sn']) != 11:
            messagebox.showerror("Invalid S/N", "S/N must be 11 characters long")
            return
        if inputs['imei'] and len(inputs['imei']) != 15:
            messagebox.showerror("Invalid IMEI", "IMEI must be 15 characters long")
            return
        if not any(inputs.values()):
            messagebox.showwarning("Input Required", "Enter at least one value to compare.")
            return

        self._result_lbl.configure(text="Fetching…", text_color="#333333")
        self._set_ui_busy(True)

        def worker():
            verdicts: list[bool] = []
            for _, key, fn in self.FIELDS:
                scanned = self._entries[key].get().strip()
                if not scanned:
                    self.after(0, lambda k=key: (
                        self._fetched_lbls[k].configure(text="— — —"),
                        self._set_status(k, "idle")
                    ))
                    continue

                fetched = fn()
                match = (scanned == fetched) and fetched != "Not Found"
                verdicts.append(match)

                status = "pass" if match else "fail"
                self.after(0, lambda k=key, v=fetched, s=status: (
                    self._fetched_lbls[k].configure(text=v),
                    self._set_status(k, s),
                ))

            self.after(0, lambda: self._finish_compare(verdicts))

        threading.Thread(target=worker, daemon=True).start()

    def _finish_compare(self, verdicts: list[bool]):
        if not verdicts:
            self._result_lbl.configure(text="PASS/FAIL", text_color="black")
        elif all(verdicts):
            self.eid_status=True
            self.update_final_result()
            self._result_lbl.configure(text="✅  PASS", text_color="#1A7A1A")
        else:
            self.eid_status=False
            self.update_final_result()
            self._result_lbl.configure(text="❌  FAIL", text_color="#BB0000")
        self._set_ui_busy(False)

def run_adb(*args):
    try:
        startupinfo = subprocess.STARTUPINFO()
        startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW
        startupinfo.wShowWindow = subprocess.SW_HIDE

        result = subprocess.check_output(
            [ADB_PATH, *args],
            startupinfo=startupinfo,
            stderr=subprocess.DEVNULL,
            timeout=10
        ).decode().strip()
        return result
    except Exception:
        return "Not Found"

arr = [] ### array which contain the values for version, rf cal etc.
def get_version() -> str:
    global arr
    arr.clear()
    try:
        text = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#12580*369#';",
            " uiautomator dump --compressed /dev/stdout | grep string")
        version = text

        if not version or version == "Not Found":
            arr.extend(["N/A", "N/A", "N/A", "N/A", "N/A"])
            return "Not Found"

        content = version.split()
        a = content[50].strip('"') if len(content) > 50 else "N/A"
        b = content[72].strip('"') if len(content) > 72 else "N/A"
        c = content[94].strip('"') if len(content) > 94 else "N/A"
        d = content[117].strip('"') if len(content) > 117 else "N/A"
        e = content[140].strip('"') if len(content) > 140 else "N/A"
        arr.extend([a, b, c, d, e])
        return "Success"
    except Exception:
        arr.extend(["Error", "Error", "Error", "Error", "Error"])
        return "Error"

def get_imei() -> str:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        result = run_adb("shell",
            "input keyevent KEYCODE_CALL; sleep 1; input text '*#06#'; "
            "uiautomator dump --compressed /dev/stdout | grep -oE '[0-9]{15}'")
        imei = result.strip()
        return imei[:15] if imei else "Not Found"
    except Exception:
        return "Not Found"

def fetch_device_info() -> dict:
    try:
        run_adb("shell", "input", "tap", "600", "2250")
        get_version()
        serial       = run_adb("get-serialno")          or "N/A"
        brand        = run_adb("shell", "getprop", "ro.product.brand")          or "N/A"
        model        = run_adb("shell", "getprop", "ro.product.model")          or "N/A"
        android_ver  = run_adb("shell", "getprop", "ro.build.version.release")  or "N/A"
        arr.append(model) #5th index
        
        battery = "N/A"
        battery_dump = run_adb("shell", "dumpsys", "battery")
        if battery_dump != "Not Found":
            for ln in battery_dump.splitlines():
                if "level:" in ln:
                    battery = ln.split(":")[-1].strip() + " %"
                    break

        wifi_mac = "N/A"
        ip_dump = run_adb("shell", "ip", "addr", "show", "wlan0")
        if ip_dump != "Not Found":
            for ln in ip_dump.splitlines():
                if "link/ether" in ln:
                    wifi_mac = ln.split()[1]
                    break

        bt_mac = (
            run_adb("shell", "settings", "get", "secure", "bluetooth_address") or "N/A"
        )

        return {
            "Serial No": serial,
            "Brand": brand.upper() if brand != "N/A" else "N/A",
            "Model": model,
            "Android Version": android_ver,
            "Battery": battery,
            "WiFi MAC": wifi_mac.upper() if wifi_mac != "N/A" else "N/A",
            "Bluetooth MAC": bt_mac.upper() if bt_mac != "N/A" else "N/A",
            "Base Version": arr[0] if len(arr) > 0 else "N/A",
            "RF Cal": arr[3] if len(arr) > 3 else "N/A",
            "HW Rev": arr[4] if len(arr) > 4 else "N/A"
        }
    except Exception as e:
        return {"Error": f"Failed fetching info: {str(e)}"}

    def extract_ui_images_from_excel(self, excel_file):
        try:
            from openpyxl import load_workbook
            os.makedirs("ref_img", exist_ok=True)
            for f in glob("ref_img/*.png"):
                try: os.remove(f)
                except Exception: pass

            workbook = load_workbook(excel_file)
            sheet = workbook.active
            images_info = []

            for img in sheet._images:
                if hasattr(img, 'anchor') and hasattr(img.anchor, '_from'):
                    cell = img.anchor._from
                    images_info.append((cell.row, cell.col, img))

            images_info.sort(key=lambda x: (x[0], x[1]))
            extracted_images = []
            
            for idx, (row, col, img) in enumerate(images_info):
                image_path = os.path.join("ref_img", f"extracted_image_{idx + 1}.png")
                with open(image_path, "wb") as img_file:
                    img_file.write(img._data())
                extracted_images.append(image_path)
            return extracted_images

        except zipfile.BadZipFile:
            messagebox.showerror("File Error", "File is not a valid zip archive (not a true .xlsx).")
            return []
        except Exception as e:
            messagebox.showerror("Error", f"Failed to extract images: {str(e)}")
            return []

    def compare_ui_images(self):
        excel_file = self.ui_file_path.get()
        if not excel_file:
            self.after(0, lambda: messagebox.showerror("Error", "Please select an Excel file for UI verification"))
            self.after(0, lambda: self._set_ui_busy(False))
            return

        try:
            import cv2
            ref_images = self.extract_ui_images_from_excel(excel_file)
            if len(ref_images) < 4:
                self.after(0, lambda: messagebox.showerror("Error", f"Extracted only {len(ref_images)} images, need 4"))
                self.after(0, lambda: self._set_ui_busy(False))
                return

            if not self.capture_device_images():
                self.after(0, lambda: self._set_ui_busy(False))
                return

            target_images = sorted(glob(os.path.abspath(os.path.join('target_img', '*.*'))))
            if len(target_images) < 4:
                self.after(0, lambda: messagebox.showerror("Error", f"Captured only {len(target_images)} screenshots"))
                self.after(0, lambda: self._set_ui_busy(False))
                return

            device_model = run_adb("shell", "getprop", "ro.product.model")
            if device_model == "Not Found": device_model = "UnknownDevice"
            os.makedirs(device_model, exist_ok=True)

            self.comparison_images = []
            results = []
            all_pass = True

            for i in range(min(4, len(ref_images), len(target_images))):
                try:
                    ref_img = cv2.imread(ref_images[i])
                    target_img = cv2.imread(target_images[i])
                    if ref_img is None or target_img is None:
                        raise ValueError("Failed to load image via cv2")

                    result_img, missing_apps, misplaced_apps = self.compare_launcher_ui(ref_img, target_img)
                    output_path = os.path.join(device_model, f"comparison_result_{i + 1}.png")
                    cv2.imwrite(output_path, result_img)
                    self.comparison_images.append(output_path) 

                    if missing_apps or misplaced_apps:
                        results.append(f"Image {i + 1}: FAIL - Missing: {len(missing_apps)}, Misplaced: {len(misplaced_apps)}")
                        all_pass = False
                    else:
                        results.append(f"Image {i + 1}: PASS")
                except Exception as e:
                    results.append(f"Image {i + 1}: Error - {str(e)}")
                    all_pass = False

            results.append("<<CLICK TO SHOW IMAGES>>")

            # Thread-safe UI update
            self.after(0, lambda: self._update_ui_results_gui(results, all_pass))

        except ImportError:
            self.after(0, lambda: messagebox.showerror("Error", "Required library (cv2) not available"))
            self.after(0, lambda: self._set_ui_busy(False))
        except Exception as e:
            self.after(0, lambda: messagebox.showerror("Error", f"UI comparison failed: {str(e)}"))

    def compare_launcher_ui(self, ref, target, similarity_thresh=0.70, position_tol=80):
        try:
            import cv2
            if ref is None or target is None:
                raise ValueError("Image not loaded")

            ref_h, ref_w = ref.shape[:2]
            tgt_h, tgt_w = target.shape[:2]
            scale_x = tgt_w / ref_w
            scale_y = tgt_h / ref_h

            result_img = target.copy()
            gray_ref = cv2.cvtColor(ref, cv2.COLOR_BGR2GRAY)
            thresh = cv2.adaptiveThreshold(gray_ref, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 11, 2)
            contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

            icon_boxes = []
            for cnt in contours:
                x, y, w, h = cv2.boundingRect(cnt)
                if h == 0: continue
                ratio = w / float(h)
                if (0.7 < ratio < 1.3 and 20 < w < 120 and 20 < h < 120):
                    icon_boxes.append((x, y, w, h))

            icon_boxes = sorted(icon_boxes, key=lambda b: (b[1], b[0]))
            orb = cv2.SIFT_create(nfeatures=3000)

            missing_apps = []
            misplaced_apps = []

            for idx, (x, y, w, h) in enumerate(icon_boxes):
                ref_icon = ref[y:y + h, x:x + w]
                kp1, des1 = orb.detectAndCompute(ref_icon, None)
                if des1 is None: continue

                expected_x = int(x * scale_x)
                expected_y = int(y * scale_y)
                search_margin = 150
                sx1 = max(0, expected_x - search_margin)
                sy1 = max(0, expected_y - search_margin)
                sx2 = min(tgt_w, expected_x + int(w * scale_x) + search_margin)
                sy2 = min(tgt_h, expected_y + int(h * scale_y) + search_margin)

                search_roi = target[sy1:sy2, sx1:sx2]
                if search_roi.size == 0: continue

                ref_resized = cv2.resize(ref_icon, (max(10, int(w * scale_x)), max(10, int(h * scale_y))))
                match = cv2.matchTemplate(search_roi, ref_resized, cv2.TM_CCOEFF_NORMED)
                _, max_val, _, max_loc = cv2.minMaxLoc(match)

                if max_val < similarity_thresh:
                    missing_apps.append({"icon_id": idx, "expected": (expected_x, expected_y)})
                    cv2.rectangle(result_img, (expected_x, expected_y), (expected_x + int(w * scale_x), expected_y + int(h * scale_y)), (0, 0, 255), 3)
                    cv2.putText(result_img, "MISSING", (expected_x, expected_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
                    continue

                found_x = sx1 + max_loc[0]
                found_y = sy1 + max_loc[1]
                if abs(found_x - expected_x) > position_tol or abs(found_y - expected_y) > position_tol:
                    misplaced_apps.append({"icon_id": idx, "expected": (expected_x, expected_y), "found": (found_x, found_y)})
                    cv2.rectangle(result_img, (found_x, found_y), (found_x + ref_resized.shape[1], found_y + ref_resized.shape[0]), (0, 165, 255), 3)
                    cv2.line(result_img, (expected_x, expected_y), (found_x, found_y), (0, 165, 255), 2)
                    cv2.putText(result_img, "MISPLACED", (found_x, found_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 165, 255), 2)

            return result_img, missing_apps, misplaced_apps

        except Exception as e:
            print(f"Error in compare_launcher_ui: {str(e)}")
            return target.copy() if target is not None else None, [], []

    def show_comparison_images(self):
        if not hasattr(self, 'comparison_images') or not self.comparison_images:
            messagebox.showinfo("Info", "No comparison images available")
            return
        
        try:
            import cv2
            windows_opened = False
            
            for image_path in self.comparison_images:
                # Convert to absolute path to guarantee it correctly finds the folder
                abs_path = os.path.abspath(image_path)
                
                if os.path.exists(abs_path):
                    img = cv2.imread(abs_path)
                    if img is not None:
                        height, width = img.shape[:2]
                        max_height = 800
                        if height > max_height:
                            scale = max_height / height
                            img = cv2.resize(img, (int(width * scale), max_height))
                        
                        window_name = os.path.basename(abs_path)
                        
                        try:
                            # Attempt to display the image
                            cv2.imshow(window_name, img)
                            windows_opened = True
                            
                            index = self.comparison_images.index(image_path)
                            cv2.moveWindow(window_name, 100 + (index % 2) * 600, 100 + (index // 2) * 500)
                            
                        except cv2.error as cv_err:
                            # Catch the specific Headless OpenCV error you are experiencing
                            if "not implemented" in str(cv_err).lower() or "rebuild the library" in str(cv_err).lower():
                                messagebox.showerror(
                                    "OpenCV Installation Error", 
                                    "Your OpenCV installation does not support UI windows.\n"
                                    "You currently have 'opencv-python-headless' installed.\n\n"
                                    "To fix this, open your terminal and run:\n"
                                    "pip uninstall opencv-python-headless\n"
                                    "pip install opencv-python"
                                )
                                return
                            else:
                                raise cv_err
                    else:
                        messagebox.showerror("Error", f"Failed to read image data from: {abs_path}")
                else:
                    messagebox.showerror("Error", f"Image not found at: {abs_path}")

            # Only pause the script if windows actually successfully opened
            if windows_opened:
                cv2.waitKey(0)
                cv2.destroyAllWindows()
                
        except ImportError:
            messagebox.showerror("Error", "Required library (cv2) not available to show images")
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")

    def _validate_entry_length(self, event, expected_length, current_key):
        entry = event.widget
        text = entry.get().strip()
        if len(text) > expected_length:
            entry.delete(expected_length, tk.END)
            return
        if len(text) == expected_length:
            current_index = list(self._entries.keys()).index(current_key)
            next_index = current_index + 1
            if next_index < len(self._entries):
                next_key = list(self._entries.keys())[next_index]
                self._entries[next_key].focus_set()

    def _on_fetch(self):
        inputs = {key: self._entries[key].get().strip() for key, *_ in [(k,) for _, k, _ in self.FIELDS]}

        if inputs['eid'] and len(inputs['eid']) != 32:
            messagebox.showerror("Invalid EID", "EID must be 32 characters long")
            return
        if inputs['sn'] and len(inputs['sn']) != 11:
            messagebox.showerror("Invalid S/N", "S/N must be 11 characters long")
            return
        if inputs['imei'] and len(inputs['imei']) != 15:
            messagebox.showerror("Invalid IMEI", "IMEI must be 15 characters long")
            return
        if not any(inputs.values()):
            messagebox.showwarning("Input Required", "Enter at least one value to compare.")
            return

        self._result_lbl.configure(text="Fetching…", text_color="#333333")
        self._set_ui_busy(True)

        def worker():
            verdicts: list[bool] = []
            for _, key, fn in self.FIELDS:
                scanned = self._entries[key].get().strip()
                if not scanned:
                    self.after(0, lambda k=key: (
                        self._fetched_lbls[k].configure(text="— — —"),
                        self._set_status(k, "idle")
                    ))
                    continue

                fetched = fn()
                match = (scanned == fetched) and fetched != "Not Found"
                verdicts.append(match)

                status = "pass" if match else "fail"
                self.after(0, lambda k=key, v=fetched, s=status: (
                    self._fetched_lbls[k].configure(text=v),
                    self._set_status(k, s),
                ))

            self.after(0, lambda: self._finish_compare(verdicts))

        threading.Thread(target=worker, daemon=True).start()

    def _finish_compare(self, verdicts: list[bool]):
        if not verdicts:
            self._result_lbl.configure(text="PASS/FAIL", text_color="black")
        elif all(verdicts):
            self.eid_status=True
            self.update_final_result()
            self._result_lbl.configure(text="✅  PASS", text_color="#1A7A1A")
        else:
            self.eid_status=False
            self.update_final_result()
            self._result_lbl.configure(text="❌  FAIL", text_color="#BB0000")
        self._set_ui_busy(False)
