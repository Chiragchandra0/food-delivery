2. adb_utils.py
This module handles all communication with your Android device.
python
import subprocess

def run_adb(command):
    try:
        # Using shell=True and capturing output
        result = subprocess.run(f"adb {command}", shell=True, capture_output=True, text=True)
        return result.stdout.strip()
    except Exception as e:
        return str(e)

def check_device():
    out = run_adb("devices")
    lines = out.split('\n')
    # If there's more than one line and the second line contains 'device', we're connected
    if len(lines) > 1 and 'device' in lines[1]:
        return True
    return False

def open_camera():
    run_adb("shell am start -a android.media.action.IMAGE_CAPTURE")

def capture_image():
    # KEYCODE_CAMERA is 27
    run_adb("shell input keyevent 27")

def pull_latest_image(save_path):
    # Get the latest file from the camera folder
    out = run_adb("shell ls -t /sdcard/DCIM/Camera")
    files = [f.strip() for f in out.split('\n') if f.lower().strip().endswith(('.jpg', '.jpeg', '.png'))]
    
    if not files:
        return False
        
    latest_file = files[0]
    run_adb(f"pull /sdcard/DCIM/Camera/{latest_file} {save_path}")
    return True

def close_camera():
    # Sends user to home screen (KEYCODE_HOME is 3)
    run_adb("shell input keyevent 3")

3. image_utils.py
This module handles cropping, combining without losing pixel data, and watermarking the final image.
python
from PIL import Image, ImageDraw, ImageFont
import os
from datetime import datetime

def calculate_crop_box(img_size, crop_left, crop_top, crop_right, crop_bottom):
    """ Converts percentage sliders into actual pixel coordinates """
    w, h = img_size
    left = int(w * (crop_left / 100))
    top = int(h * (crop_top / 100))
    right = int(w * (1 - (crop_right / 100)))
    bottom = int(h * (1 - (crop_bottom / 100)))
    
    # Prevent crashing if sliders cross each other
    if left >= right: left = right - 1
    if top >= bottom: top = bottom - 1
        
    return (left, top, right, bottom)

def combine_side_by_side(img1_path, img2_path, output_path, crop1, crop2):
    try:
        img1 = Image.open(img1_path)
        img2 = Image.open(img2_path)

        # Apply crops based on UI slider percentages
        box1 = calculate_crop_box(img1.size, *crop1)
        box2 = calculate_crop_box(img2.size, *crop2)
        
        img1 = img1.crop(box1)
        img2 = img2.crop(box2)

        # Maintain original pixels, calculate new canvas size
        w1, h1 = img1.size
        w2, h2 = img2.size
        
        new_width = w1 + w2
        new_height = max(h1, h2)
        
        combined = Image.new('RGB', (new_width, new_height))
        
        # Paste side by side
        combined.paste(img1, (0, 0))
        combined.paste(img2, (w1, 0))
        
        combined.save(output_path, quality=100)
        return True
    except Exception as e:
        print(f"Error combining images: {e}")
        return False

def add_watermark_and_save(input_path, name, user_id, output_dir):
    try:
        img = Image.open(input_path)
        draw = ImageDraw.Draw(img)
        
        # Fallback to default font if arial isn't found
        try:
            font = ImageFont.truetype("arial.ttf", 80)
        except IOError:
            font = ImageFont.load_default()

        text = f"{name} - {user_id}"
        
        # Calculate text positioning (Bottom Right)
        bbox = draw.textbbox((0, 0), text, font=font)
        text_width = bbox[2] - bbox[0]
        text_height = bbox[3] - bbox[1]
        
        width, height = img.size
        x = width - text_width - 60  # 60px padding from right
        y = height - text_height - 60 # 60px padding from bottom
        
        # Draw stroke for visibility against any background
        stroke = 4
        draw.text((x, y), text, font=font, fill="white", stroke_width=stroke, stroke_fill="black")
        
        # Save as Name + Datetime
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{name}_{timestamp}.png"
        save_path = os.path.join(output_dir, filename)
        
        img.save(save_path, format="PNG")
        return save_path
    except Exception as e:
        print(f"Error saving watermarked image: {e}")
        return None

How to use the non-destructive cropping:
Because building a custom free-draw bounding box inside a Tkinter canvas is complex, I added four Crop Percentage Sliders (Left, Top, Right, Bottom) beneath each captured image preview.
1. Capture your images.
2. If you only want the center of an image, adjust the left/right or top/bottom sliders (e.g., setting the first slider to 10 chops 10% off the left edge).
3. Click Combine Images. The original full-resolution files are dynamically cropped and stitched without altering their DPI or pixel density, resulting in a pristine output.
4. Click Save & Watermark to enter the Name and ID.
